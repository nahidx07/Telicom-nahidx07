rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // ইউজার শুধু নিজের প্রোফাইল এবং ব্যালেন্স দেখতে পারবে
    match /users/{userId} {
      allow read: if request.auth != null && request.auth.uid == userId;
      allow write: if false; // ব্যালেন্স শুধু এডমিন বা ক্লাউড ফাংশন থেকে আপডেট হবে
    }
    
    // সবাই প্যাকেজ দেখতে পারবে
    match /packages/{packageId} {
      allow read: if true;
      allow write: if get(/databases/$(database)/documents/users/$(request.auth.uid)).data.isAdmin == true;
    }
    
    // অর্ডার ক্রিয়েট করার নিয়ম
    match /orders/{orderId} {
      allow create: if request.auth != null;
      allow read: if request.auth != null && (resource.data.userId == request.auth.uid || get(/databases/$(database)/documents/users/$(request.auth.uid)).data.isAdmin == true);
    }
  }
}
